ScriptName FindOffset
# script to find proper offset value to run Z_byV2
# assume speciment is ON the eucentricity 

## Eucentric Z
##
Eucentricity 3
ReportStageXYZ 
Z0 = $repVal3

## now find the offset
offset = -288.0             # good for -300um V vs R
Loop 10
CallFunction MyFuncs::Z_byV2 $offset

ReportStageXYZ 
Z = $repVal3
diffZ = $Z - $Z0
echo $diffZ
   If  ABS $diffZ < 0.5 
      echo >>> Found "offset" is $offset
      echo >>> run "Z_byZ2 $offset" 
      exit
   Else 
      offset = $offset + $diffZ
   Endif 
EndLoop
